deep copy and shallow copy 
	

shallow copy 
	if u change the nested object value original value also change 
		
	original = [[1, 2], [3, 4]]
shallow = copy.copy(original)
	
	shallow[0][0] = 100
print(original)  # [[100, 2], [3, 4]]  ← original is affected
	
deep copy 	
		
	it wont affect the original copy 
	
	
	import copy

original = [[1, 2], [3, 4]]
deep = copy.deepcopy(original)

deep[0][0] = 100
print(original)  # [[1, 2], [3, 4]]  ← original NOT affected

pop and remove
	
	pop --->default remove last value 
	
	remove ----> remove exact value 



lst = [10, 20, 30, 20]

x = lst.pop()    # removes last element
print(x)         # 20
print(lst)       # [10, 20, 30]

lst.remove(20)   # removes first 20
print(lst)       # [10, 30]
